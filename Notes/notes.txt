 using Newtonsoft.Json;

[Serializable]
public class HollywoodMovie
{
    public string moviename;
    public int ticketprice;


    public void movielist()
    {
        List<HollywoodMovie> ls = new List<HollywoodMovie>();
        ls.Add(new HollywoodMovie()
        {
            moviename = "Tenet",
            ticketprice = 250
        });
        ls.Add(new HollywoodMovie()
        {
            moviename = "Jurassic park",
            ticketprice = 350
        });
        ls.Add(new HollywoodMovie()
        {
            moviename = "Gone girl",
            ticketprice = 150
        });

        //JSON Serialization
        string memory;
        memory = JsonConvert.SerializeObject(ls);
        Console.WriteLine(memory);

        //JSON Deserialization

        List<HollywoodMovie> des = JsonConvert.DeserializeObject<List<HollywoodMovie>>(memory);
        foreach(var i in des)
        {
            Console.WriteLine("Movie Name :" + i.moviename);
            Console.WriteLine("Ticket price :" + i.ticketprice);
        }
    }





}

public class Program
{
public static void Main(string[] args)
    {
